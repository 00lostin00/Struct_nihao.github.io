#include <bits/stdc++.h>
using namespace std;
const int N = 101001;
#define LL long long
int n, m, K;
LL ans;
const int MOD = 1e9 + 7;
LL dp[200][200][30][30];
int main()
{
    cin >> n >> m >> K;
    dp[1][0][1][0] = 1;
    dp[0][1][0][1] = 1;
    for (int i = 0; i <= n; i++)
    {
        for (int j = 0; j <= m; j++)
        {
            for (int q = 0; q <= K; q++)
            {
                for (int w = 0; w <= K; w++)
                {
                    if (q >= 1 && i >= 1)
                    {
                        dp[i][j][q][w] = (dp[i][j][q][w] + dp[i - 1][j][q - 1][w + 1]) % MOD;
                        if (w == 0)
                            dp[i][j][q][w] = (dp[i][j][q][w] + dp[i - 1][j][q - 1][0]) % MOD;
                    }
                    if (w >= 1 && j >= 1)
                    {
                        dp[i][j][q][w] = (dp[i][j][q][w] + dp[i][j - 1][q + 1][w - 1]) % MOD;
                        if (q == 0)
                            dp[i][j][q][w] = (dp[i][j][q][w] + dp[i][j - 1][0][w - 1]) % MOD;
                    }
                }
            }
        }
    }
    for (int i = 0; i <= K; i++)
    {
        for (int j = 0; j <= K; j++)
        {
            ans = (ans + dp[n][m][i][j]) % MOD;
        }
    }
    cout << ans;
    return 0;
}

#include <bits/stdc++.h>
using namespace std;
const int N = 101001;
const int mod = 1e9 + 7;
typedef long long LL;
LL dp[200][200][30][30];
int n, m, K;
LL dfs(int i, int j, int q, int w)
{
    if (i == n && j == m)
        return 1;
    if (dp[i][j][q][w] != -1)
        return dp[i][j][q][w];
    LL tmp = 0;
    if (i < n && q < K)
    {
        tmp = (tmp + dfs(i + 1, j, q + 1, max(0, w - 1))) % mod;
    }
    if (j < m && w < K)
    {
        tmp = (tmp + dfs(i, j + 1, max(0, q - 1), w + 1)) % mod;
    }
    return dp[i][j][q][w] = tmp;
}
int main()
{
    cin >> n >> m >> K;
    for (int i = 0; i <= n; i++)
        for (int j = 0; j <= m; j++)
            for (int q = 0; q <= K; q++)
                for (int w = 0; w <= K; w++)
                    dp[i][j][q][w] = -1;
    cout << dfs(0, 0, 0, 0);
    return 0;
}
